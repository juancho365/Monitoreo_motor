// ===== CRC16 MODBUS =====
uint16_t modbusCRC(uint8_t *buf, int len) {
  uint16_t crc = 0xFFFF;
  for (int pos = 0; pos < len; pos++) {
    crc ^= (uint16_t)buf[pos];
    for (int i = 0; i < 8; i++) {
      if (crc & 0x0001) { crc >>= 1; crc ^= 0xA001; }
      else { crc >>= 1; }
    }
  }
  return crc;
}

void enviarConsulta() {
  uint8_t frame[8];

  frame[0] = SLAVE_ID;
  frame[1] = 0x03;
  frame[2] = 0x00;
  frame[3] = 0x00;
  frame[4] = 0x00;
  frame[5] = 0x01;

  uint16_t crc = modbusCRC(frame, 6);
  frame[6] = crc & 0xFF;
  frame[7] = crc >> 8;

  String payload = "";
  for (int i = 0; i < 8; i++) {
    if (frame[i] < 16) payload += "0";
    payload += String(frame[i], HEX);
  }

  String cmd = "AT+SEND=" + String(SLAVE_ID) + "," +
               String(payload.length()) + "," + payload;

  lora.println(cmd);
  Serial.println("Consulta enviada a Esclavo LoRa");
}

void procesarRespuesta(String line) {
  int first = line.indexOf(',');
  int second = line.indexOf(',', first + 1);
  int third = line.indexOf(',', second + 1);
  int fourth = line.indexOf(',', third + 1); 
  
  // Extraer los datos Modbus
  String dataHex = line.substring(second + 1, third);
  
  // Guardar el RSSI real del enlace LoRa
  if(third != -1 && fourth != -1) {
    ultimoRSSI = line.substring(third + 1, fourth).toInt();
  }

  int len = dataHex.length() / 2;
  uint8_t frame[64];

  for (int i = 0; i < len; i++) {
    String byteStr = dataHex.substring(i * 2, i * 2 + 2);
    frame[i] = strtol(byteStr.c_str(), NULL, 16);
  }

  uint16_t crcRx = frame[len - 2] | (frame[len - 1] << 8);
  uint16_t crcCalc = modbusCRC(frame, len - 2);

  if (crcRx != crcCalc) {
    Serial.println("CRC ERROR - Trama corrupta");
    return;
  }

  if (frame[1] == 0x03 && frame[2] == 0x02) {
    uint16_t temp = (frame[3] << 8) | frame[4];
    float tempC = temp / 10.0;

    Serial.print(">>>> TEMPERATURA RECIBIDA: "); Serial.print(tempC);
    Serial.print(" °C | RSSI: "); Serial.print(ultimoRSSI); Serial.println(" dBm");

    mostrarTemperatura(tempC);
    ultimaTemperatura = tempC;  
    
    // ¡NUEVO!: Actualizamos el reloj porque tuvimos una lectura exitosa
    lastSuccessfulRead = millis();
  }
}

void setup() {
  Serial.begin(115200); 
  
  // 1. Iniciar OLED primero
  Wire.begin(SDA_PIN, SCL_PIN);
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {

    while (true); 
  }
  
  // 2. Mostrar mensaje de bienvenida
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(0, 10);
  display.println("Iniciando ESP32...");
  display.display();
  delay(1000); 

  // 3. Conectar a WiFi con feedback visual
  display.clearDisplay();
  display.setCursor(0, 10);
  display.println("Conectando WiFi:");
  display.println(ssid);
  display.display();
  
  WiFi.begin(ssid, password);
  Serial.print("Conectando a WiFi ");
  
  int intentos = 0;
  // Intenta conectar 20 veces (10 segundos)
  while (WiFi.status() != WL_CONNECTED && intentos < 20) {
    delay(500);
    Serial.print(".");
    display.print("."); // Muestra los puntos en la OLED
    display.display();
    intentos++;
  }

  // 4. Verificar si logró conectar
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\nWiFi Conectado!");
    display.println("\nWiFi OK!");
    display.display();
    delay(1500);
  } else {
    // Si falla, muestra error y reinicia la placa
    Serial.println("\nError: No hay WiFi. Reiniciando...");
    display.println("\nFallo! Reiniciando");
    display.display();
    delay(3000);
    ESP.restart(); 
  }
  
  // 5. Configurar e Iniciar Firebase
  Serial.println("Conectando a Firebase...");
  config.api_key = API_KEY;
  config.database_url = DATABASE_URL;

  // ACTIVAMOS EL MODO PRUEBA PARA SALTARNOS EL BLOQUEO (ERROR 503)
  config.signer.test_mode = true; 
  signupOK = true; 
  Serial.println("Firebase configurado en Modo Prueba (Sin Autenticación)");

  config.token_status_callback = tokenStatusCallback;
  Firebase.begin(&config, &auth);
  Firebase.reconnectWiFi(true); 

  // 6. Configurar LoRa
  lora.begin(9600, SERIAL_8N1, 16, 17); 
  
  // Inicializar reloj de seguridad
  lastSuccessfulRead = millis();

  // 7. Pantalla lista
  display.clearDisplay();
  display.setCursor(0, 10);
  display.println("Esperando datos");
  display.println("por LoRa...");
  display.display();
}

void loop() {
  // Consulta de Modbus al Esclavo
  if (millis() - lastQuery > interval) {
    lastQuery = millis();
    enviarConsulta();
  }
  
  // Escuchar respuestas LoRa
  while (lora.available()) {
    String line = lora.readStringUntil('\n');
    line.trim();
    
    if (line.length() > 0) {
      Serial.println("LoRa Escucho: " + line); 
      if (line.startsWith("+RCV=")) {
        procesarRespuesta(line);
      }
    }
  }

 // Publicar a Firebase (Cada 10 segundos)
  if (Firebase.ready() && signupOK && (millis() - lastFirebaseUpdate > firebaseInterval)) {
    lastFirebaseUpdate = millis();
    
    // ==============================================================================
    // ¡LÓGICA DE DESCONEXIÓN!: Si pasan más de 15s sin datos de LoRa, asumimos error
    // ==============================================================================
    if (millis() - lastSuccessfulRead > timeoutEsclavo) {
        ultimaTemperatura = -999.0; // Forzamos temperatura irreal de desconexión
        mostrarErrorConexion();     // Mostramos aviso en la pantalla física OLED
        Serial.println("¡ALERTA! Timeout LoRa/Modbus. Reportando -999 a Firebase.");
    }
    
    // Solo subimos datos si ya recibimos una lectura (o si activamos la desconexión -999)
    if (ultimaTemperatura != 0.0) {
      
      FirebaseJson json;
      json.set("Temperatura", ultimaTemperatura);
      json.set("temperatura", ultimaTemperatura);
      json.set("Temp", ultimaTemperatura);
      json.set("temp", ultimaTemperatura);
      
      json.set("RSSI", ultimoRSSI);
      json.set("rssi", ultimoRSSI);
      
      if (Firebase.RTDB.pushJSON(&fbdo, "Nodo1", &json)){
        Serial.println("Firebase: Nuevo registro añadido.");
      } else {
        Serial.println("Error Firebase: " + fbdo.errorReason());
      }
      
    } else {
      Serial.println("Firebase: Esperando dato LoRa real antes de subir...");
    }
  }
}
